#include <stdio.h>

int main()
{
    int size;
    printf("Enter the size\n");
    scanf("%d", &size);

    int array[size];
    int dup[size];
    int count = 0;

    printf("Enter %d elements:\n", size);
    for (int i = 0; i < size; i++)
        scanf("%d", &array[i]);

    for (int i = 0; i < size; i++)
    {
        int is_unique = 1;  // Assume it's unique
        for (int j = 0; j < count; j++)
        {
            if (array[i] == dup[j])
            {
                is_unique = 0;
                break;
            }
        }
        if (is_unique)
        {
            dup[count++] = array[i];
        }
    }

    printf("Unique elements are:\n");
    for (int i = 0; i < count; i++)
        printf("%d ", dup[i]);

    return 0;
}
